package main

import (
	"fmt"
)

func main() {
	nums := []int{3, 4, 5, 6, 7, 8}
	fmt.Println(GetSumOfSquares(nums))
}

func GetSumOfSquares(nums []int) int {

	ch := make(chan int)

	totalSum := 0
	for _, num := range nums {
		go func(x int) {
			ch <- x * x
		}(num)
	}

	for range nums {
		squared := <-ch
		fmt.Println("Got :", squared)
		totalSum += squared
	}
	return totalSum
}
