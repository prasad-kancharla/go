package advanced

import (
	"fmt"
	"testing"
)

// file name should have suffix of _test
func Add(a, b int) int {
	return a + b
	// function names should have Test as prefix
}

// command to run tests is
// cd advanced
// go test testing_test.go
func TestAdd(t *testing.T) {
	result := Add(2, 3)
	expected := 5

	if result != expected {
		t.Errorf("Add(2,3) = %d, want %d", result, expected)
	}
}

func TestAddTableDriven(t *testing.T) {
	tests := []struct{ a, b, expected int }{
		{0, 0, 0},
		{2, 3, 5},
		{1, -1, 0},
	}

	for _, test := range tests {
		result := Add(test.a, test.b)
		if result != test.expected {
			t.Errorf("Add(%d, %d) = %d; want %d", test.a, test.b, result, test.expected)
		}
	}
}

func TestAddSubtests(t *testing.T) {
	tests := []struct{ a, b, expected int }{
		{9, 3, 12},
		{3, -4, -1},
		{100, 8, 108},
	}

	for i, test := range tests {
		// we can give name to each sub test
		t.Run(fmt.Sprintf("Test %d", i+1), func(t *testing.T) {
			result := Add(test.a, test.b)
			if result != test.expected {
				t.Errorf("Result is %d, want %d", result, test.expected)
			}
		})
	}
}
